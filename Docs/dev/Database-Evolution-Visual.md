
### 3. Soft Deletes

```sql
-- Tier 1: User deletes report
UPDATE snapshots SET deleted_at = NOW() WHERE id = 'xyz';

-- Tier 2: Compliance requirement - keep deleted data 30 days
SELECT * FROM snapshots WHERE deleted_at < NOW() - INTERVAL '30 days';

-- Hard delete after 30 days
DELETE FROM snapshots WHERE deleted_at < NOW() - INTERVAL '30 days';
```

### Tier 2 (14 Tables)

```

SELECT 
  s.*,
  ss.frequency as schedule_frequency,
  sh.changes_detected as last_changes
FROM snapshots s
LEFT JOIN snapshot_schedules ss ON s.schedule_id = ss.id
LEFT JOIN snapshot_history sh ON sh.current_snapshot_id = s.id
WHERE s.user_id = 'xyz'
ORDER BY s.created_at DESC;
```
